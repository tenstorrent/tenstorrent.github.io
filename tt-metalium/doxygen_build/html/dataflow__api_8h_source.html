229template <uint32_t tile_hw = 1024>
230FORCE_INLINE constexpr static std::uint32_t MUL_WITH_TILE_SIZE(uint format, uint index) {
231 constexpr uint8_t datum_shift = (tile_hw == 1024) ? 10
232 : (tile_hw == 512) ? 9
233 : (tile_hw == 256) ? 8
234 : (tile_hw == 128) ? 7
235 : (tile_hw == 64) ? 6
236 : (tile_hw == 32) ? 5
237 : (tile_hw == 16) ? 4
238 : 10;
239
240 constexpr uint8_t exp_shift = (tile_hw == 1024) ? 6
241 : (tile_hw == 512) ? 5
242 : (tile_hw == 256) ? 4
243 : (tile_hw == 128) ? 4
244 : (tile_hw == 64) ? 4
245 : (tile_hw == 32) ? 4
246 : (tile_hw == 16) ? 4
247 : 6;
248 switch (format & 0x1F) {
249 case ((uint8_t)DataFormat::UInt8): return (index << datum_shift);
250 case ((uint8_t)DataFormat::UInt16):
251 case ((uint8_t)DataFormat::Float16):
252 case ((uint8_t)DataFormat::Float16_b): return (index << (datum_shift + 1));
253 case ((uint8_t)DataFormat::Int32):
254 case ((uint8_t)DataFormat::UInt32):
255 case ((uint8_t)DataFormat::Float32): return (index << (datum_shift + 2));
256 case ((uint8_t)DataFormat::Bfp2):
257 case ((uint8_t)DataFormat::Bfp2_b): return ((index << (datum_shift - 2)) + (index << (exp_shift)));
258 case ((uint8_t)DataFormat::Bfp4):
259 case ((uint8_t)DataFormat::Bfp4_b): return ((index << (datum_shift - 1)) + (index << (exp_shift)));
260 case ((uint8_t)DataFormat::Bfp8):
261 case ((uint8_t)DataFormat::Bfp8_b):
262 // Keep default as Bfp8?
263 default: return ((index << datum_shift) + (index << (exp_shift)));
264 };
265}

291void cb_push_back(const int32_t operand, const int32_t num_pages) {
292 uint32_t num_words = num_pages * get_local_cb_interface(operand).fifo_page_size;
293
294 volatile tt_reg_ptr uint32_t* pages_received_ptr = get_cb_tiles_received_ptr(operand);
295 pages_received_ptr[0] += num_pages;
296
297 get_local_cb_interface(operand).fifo_wr_ptr += num_words;
298
299 // this will basically reset fifo_wr_ptr to fifo_addr -- no other wrap is legal
300 // producer always writes into contiguous memory, it cannot wrap
301 ASSERT(get_local_cb_interface(operand).fifo_wr_ptr <= get_local_cb_interface(operand).fifo_limit);
302 if (get_local_cb_interface(operand).fifo_wr_ptr == get_local_cb_interface(operand).fifo_limit) {
303 // TODO: change this to fifo_wr_ptr
304 get_local_cb_interface(operand).fifo_wr_ptr -= get_local_cb_interface(operand).fifo_size;
305 }
306}

443FORCE_INLINE
444bool cb_pages_reservable_at_back(int32_t operand, int32_t num_pages) {
445 uint32_t pages_acked_ptr = (uint32_t)get_cb_tiles_acked_ptr(operand);
446
447 // while the producer (write-side interface) is waiting for space to free up "tiles_pushed" is not changing
448 // "tiles_pushed" is updated by the producer only when the tiles are pushed
449 uint32_t pages_received = get_cb_tiles_received_ptr(operand)[0];
450
451 // uint16_t's here because Tensix updates the val at tiles_acked_ptr as uint16 in llk_pop_tiles
452 // TODO: I think we could have TRISC update tiles_acked_ptr, and we wouldn't need uint16 here
453 uint16_t pages_acked = (uint16_t)reg_read(pages_acked_ptr);
454#ifdef ARCH_GRAYSKULL
455 // The following test slows down by 5% when removing the barrier
456 // TODO(pgk) investigate GS arbiter WAR in compiler, is this fixing an issue there?
457 // models/experimental/stable_diffusion/tests/test_perf_unbatched_stable_diffusion.py::test_perf_bare_metal
458 volatile uint32_t local_mem_barrier = pages_acked;
459#endif
460 uint16_t free_space_pages_wrap = get_local_cb_interface(operand).fifo_num_pages - (pages_received - pages_acked);
461 return num_pages <= static_cast<int32_t>(free_space_pages_wrap);
462}

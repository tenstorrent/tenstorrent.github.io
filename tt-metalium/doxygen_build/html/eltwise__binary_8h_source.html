183template<bool full_init = false, EltwiseBinaryType eltwise_binary_op_type = ELWADD>
184ALWI void binary_op_specific_init() // TODO(AP): better naming
185{
186 if constexpr (full_init) {
187 if constexpr (eltwise_binary_op_type == ELWADD) {
188 add_tiles_init();
189 } else if constexpr (eltwise_binary_op_type == ELWSUB) {
190 sub_tiles_init();
191 } else if constexpr (eltwise_binary_op_type == ELWMUL) {
192 mul_tiles_init();
193 }
194 } else {
195 if constexpr (eltwise_binary_op_type == ELWADD) {
196 add_tiles_init_nof();
197 } else if constexpr (eltwise_binary_op_type == ELWSUB) {
198 sub_tiles_init_nof();
199 } else if constexpr (eltwise_binary_op_type == ELWMUL) {
200 mul_tiles_init_f();
201 }
202 }
203}

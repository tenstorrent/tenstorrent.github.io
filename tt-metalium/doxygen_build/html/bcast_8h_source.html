128template <EltwiseBinaryType tBcastOp, BroadcastType tBcastDim>
129void init_bcast(uint32_t icb0, uint32_t icb1, uint32_t ocb = 16) {
130 if constexpr (tBcastOp == ELWMUL) {
131 MATH((llk_math_eltwise_binary_init<tBcastOp, tBcastDim, MATH_FIDELITY>()));
132 } else {
133 MATH((llk_math_eltwise_binary_init<tBcastOp, tBcastDim>()));
134 }
135
136 UNPACK((llk_unpack_AB_hw_configure_disaggregated<DST_ACCUM_MODE>(icb0, icb1)));
137 UNPACK((llk_unpack_AB_init<tBcastDim>(icb0, icb1)));
138 // TODO(AP): running this specific init after common AB init causes a hang
139
140 // clone of general init for AB TODO(AP): commonize
141 // UNPACK(( llk_unpack_AB_init<BroadcastType::NONE>() ));
142 // UNPACK(( llk_unpack_AB_hw_configure_disaggregated<BroadcastType::NONE>(icb0, icb1) ));
143
144 PACK((llk_pack_hw_configure_disaggregated<false, DST_ACCUM_MODE>(ocb)));
145 PACK((llk_pack_init(ocb)));
146 PACK((llk_pack_dest_init<false, DST_ACCUM_MODE>()));
147
148 MATH((llk_math_pack_sync_init<DST_ACCUM_MODE>()));
149 MATH((llk_math_hw_configure_disaggregated(icb0, icb1)));
150}

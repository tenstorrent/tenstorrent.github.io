127ALWI void tilize_block(uint32_t icb, uint32_t block, uint32_t ocb)
128{
129 UNPACK(( llk_unpack_tilize_block(icb, block) ));
130
131 for (uint32_t t = 0; t < block; t++) {
132 // Acquire dst
133 MATH(( llk_math_wait_for_dest_available() ));
134 PACK(( llk_packer_wait_for_math_done() ));
135
136 // Datacopy
137 MATH(( llk_math_eltwise_unary_datacopy<A2D, BroadcastType::NONE>(0 /*dst index*/) ));
138 PACK(( llk_pack<false, false >(0 /*tile index*/, ocb) ));
139
140 // Release dest
141 MATH(( llk_math_dest_section_done<DST_ACCUM_MODE>() ));
142 PACK(( llk_pack_dest_section_done<DST_ACCUM_MODE>() ));
143 }

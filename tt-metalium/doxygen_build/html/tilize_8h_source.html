28 MATH(( llk_math_pack_sync_init<DST_ACCUM_MODE>() ));

34
35 UNPACK(( llk_unpack_tilize_hw_configure_disaggregated<DST_ACCUM_MODE>(icb) ));
36 UNPACK(( llk_unpack_tilize_init(icb, block) ));

124ALWI void tilize_block(uint32_t icb, uint32_t block, uint32_t ocb)
125{
126 UNPACK(( llk_unpack_tilize_block(icb, block) ));
127
128 for (uint32_t t = 0; t < block; t++) {
129 // Acquire dst
130 MATH(( llk_math_wait_for_dest_available() ));
131 PACK(( llk_packer_wait_for_math_done() ));
132
133 // Datacopy
134 MATH(( llk_math_eltwise_unary_datacopy<A2D, BroadcastType::NONE>(0 /*dst index*/) ));
135 PACK(( llk_pack<false, false >(0 /*tile index*/, ocb) ));
136
137 // Release dest
138 MATH(( llk_math_dest_section_done<DST_ACCUM_MODE>() ));
139 PACK(( llk_pack_dest_section_done<DST_ACCUM_MODE>() ));
140 }
141}

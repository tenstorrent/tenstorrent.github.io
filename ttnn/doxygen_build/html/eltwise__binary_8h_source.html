182template <bool full_init = false, EltwiseBinaryType eltwise_binary_op_type = ELWADD>
183ALWI void binary_op_specific_init() // TODO(AP): better naming
184{
185 if constexpr (full_init) {
186 if constexpr (eltwise_binary_op_type == ELWADD) {
187 add_tiles_init();
188 } else if constexpr (eltwise_binary_op_type == ELWSUB) {
189 sub_tiles_init();
190 } else if constexpr (eltwise_binary_op_type == ELWMUL) {
191 mul_tiles_init();
192 }
193 } else {
194 if constexpr (eltwise_binary_op_type == ELWADD) {
195 add_tiles_init_nof();
196 } else if constexpr (eltwise_binary_op_type == ELWSUB) {
197 sub_tiles_init_nof();
198 } else if constexpr (eltwise_binary_op_type == ELWMUL) {
199 mul_tiles_init_f();
200 }
201 }
202}

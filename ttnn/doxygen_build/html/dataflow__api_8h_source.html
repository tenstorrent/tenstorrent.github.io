244template <uint32_t tile_hw = 1024>
245FORCE_INLINE constexpr static std::uint32_t MUL_WITH_TILE_SIZE(uint format, uint index) {
246 constexpr uint8_t datum_shift = (tile_hw == 1024) ? 10 :
247 (tile_hw == 512) ? 9 :
248 (tile_hw == 256) ? 8 :
249 (tile_hw == 128) ? 7 :
250 (tile_hw == 64) ? 6 :
251 (tile_hw == 32) ? 5 : 10;
252
253 constexpr uint8_t exp_shift = (tile_hw == 1024) ? 6 :
254 (tile_hw == 512) ? 5 :
255 (tile_hw == 256) ? 4 :
256 (tile_hw == 128) ? 3 :
257 (tile_hw == 64) ? 2 :
258 (tile_hw == 32) ? 1 : 6;
259 switch (format & 0x1F) {
260 case ((uint8_t)DataFormat::UInt8): return (index << datum_shift);
261 case ((uint8_t)DataFormat::UInt16):
262 case ((uint8_t)DataFormat::Float16):
263 case ((uint8_t)DataFormat::Float16_b): return (index << (datum_shift + 1));
264 case ((uint8_t)DataFormat::Int32):
265 case ((uint8_t)DataFormat::UInt32):
266 case ((uint8_t)DataFormat::Float32): return (index << (datum_shift + 2));
267 case ((uint8_t)DataFormat::Bfp2):
268 case ((uint8_t)DataFormat::Bfp2_b): return ((index << (datum_shift - 2)) + (index << (exp_shift)));
269 case ((uint8_t)DataFormat::Bfp4):
270 case ((uint8_t)DataFormat::Bfp4_b): return ((index << (datum_shift - 1)) + (index << (exp_shift)));
271 case ((uint8_t)DataFormat::Bfp8):
272 case ((uint8_t)DataFormat::Bfp8_b):
273 // Keep default as Bfp8?
274 default: return ((index << datum_shift) + (index << (exp_shift)));
275 };

300FORCE_INLINE
301void cb_push_back(const int32_t operand, const int32_t num_pages) {
302
303 uint32_t num_words = num_pages * cb_interface[operand].fifo_page_size;
304
305 volatile tt_reg_ptr uint32_t* pages_received_ptr = get_cb_tiles_received_ptr(operand);
306 pages_received_ptr[0] += num_pages;
307
308 cb_interface[operand].fifo_wr_ptr += num_words;
309
310 // this will basically reset fifo_wr_ptr to fifo_addr -- no other wrap is legal
311 // producer always writes into contiguous memory, it cannot wrap
312 ASSERT(cb_interface[operand].fifo_wr_ptr <= cb_interface[operand].fifo_limit);
313 if (cb_interface[operand].fifo_wr_ptr == cb_interface[operand].fifo_limit) {
314 // TODO: change this to fifo_wr_ptr
315 cb_interface[operand].fifo_wr_ptr -= cb_interface[operand].fifo_size;
316 }
317}

453FORCE_INLINE
454bool cb_pages_reservable_at_back(int32_t operand, int32_t num_pages) {
455 uint32_t pages_acked_ptr = (uint32_t)get_cb_tiles_acked_ptr(operand);
456
457 // while the producer (write-side interface) is waiting for space to free up "tiles_pushed" is not changing
458 // "tiles_pushed" is updated by the producer only when the tiles are pushed
459 uint32_t pages_received = get_cb_tiles_received_ptr(operand)[0];
460
461 invalidate_l1_cache();
462 // uint16_t's here because Tensix updates the val at tiles_acked_ptr as uint16 in llk_pop_tiles
463 // TODO: I think we could have TRISC update tiles_acked_ptr, and we wouldn't need uint16 here
464 uint16_t pages_acked = (uint16_t)reg_read(pages_acked_ptr);
465#ifdef ARCH_GRAYSKULL
466 // The following test slows down by 5% when removing the barrier
467 // TODO(pgk) investigate GS arbiter WAR in compiler, is this fixing an issue there?
468 // models/experimental/stable_diffusion/tests/test_perf_unbatched_stable_diffusion.py::test_perf_bare_metal
469 volatile uint32_t local_mem_barrier = pages_acked;
470#endif
471 uint16_t free_space_pages_wrap = cb_interface[operand].fifo_num_pages - (pages_received - pages_acked);
472 return num_pages <= static_cast<int32_t>(free_space_pages_wrap);
473}

189FORCE_INLINE
190constexpr static std::int32_t GET_TILE_SIZE(uint format) {
191 switch (format & 0x1F) {
192 case ((uint8_t)DataFormat::Float16_b): return ((2048));
193 case ((uint8_t)DataFormat::Float16): return ((2048));
194
195 case ((uint8_t)DataFormat::UInt8): return ((1024));
196 case ((uint8_t)DataFormat::UInt16): return ((2048));
197
198 case ((uint8_t)DataFormat::Bfp8):
199 case ((uint8_t)DataFormat::Bfp8_b): return ((1024) + (64));
200
201 case ((uint8_t)DataFormat::Int32):
202 case ((uint8_t)DataFormat::UInt32):
203 case ((uint8_t)DataFormat::Float32): return ((4096));
204
205 case ((uint8_t)DataFormat::Bfp4):
206 case ((uint8_t)DataFormat::Bfp4_b): return ((512) + (64));
207
208 case ((uint8_t)DataFormat::Bfp2):
209 case ((uint8_t)DataFormat::Bfp2_b): return ((256) + (64));
210 default: return ((1024) + (64));
211 };
212}

214FORCE_INLINE
215constexpr static std::uint32_t MUL_WITH_TILE_SIZE(uint format, uint index) {
216 switch (format & 0x1F) {
217 case ((uint8_t)DataFormat::UInt8): return (index << 10);
218 case ((uint8_t)DataFormat::UInt16):
219 case ((uint8_t)DataFormat::Float16):
220 case ((uint8_t)DataFormat::Float16_b): return (index << 11);
221 case ((uint8_t)DataFormat::Int32):
222 case ((uint8_t)DataFormat::UInt32):
223 case ((uint8_t)DataFormat::Float32): return (index << 12);
224 case ((uint8_t)DataFormat::Bfp2):
225 case ((uint8_t)DataFormat::Bfp2_b): return ((index << 8) + (index << 6));
226 case ((uint8_t)DataFormat::Bfp4):
227 case ((uint8_t)DataFormat::Bfp4_b): return ((index << 9) + (index << 6));
228 case ((uint8_t)DataFormat::Bfp8):
229 case ((uint8_t)DataFormat::Bfp8_b):
230 // Keep default as Bfp8?
231 default: return ((index << 10) + (index << 6));
232 };
233}

257FORCE_INLINE
258void cb_push_back(const int32_t operand, const int32_t num_pages) {
259
260 uint32_t num_words = num_pages * cb_interface[operand].fifo_page_size;
261
262 volatile tt_reg_ptr uint32_t* pages_received_ptr = get_cb_tiles_received_ptr(operand);
263 pages_received_ptr[0] += num_pages;
264
265 cb_interface[operand].fifo_wr_ptr += num_words;
266
267 // this will basically reset fifo_wr_ptr to fifo_addr -- no other wrap is legal
268 // producer always writes into contiguous memory, it cannot wrap
269 ASSERT(cb_interface[operand].fifo_wr_ptr <= cb_interface[operand].fifo_limit);
270 if (cb_interface[operand].fifo_wr_ptr == cb_interface[operand].fifo_limit) {
271 // TODO: change this to fifo_wr_ptr
272 cb_interface[operand].fifo_wr_ptr -= cb_interface[operand].fifo_size;
273 }
274}

400FORCE_INLINE
401void cb_reserve_back(int32_t operand, int32_t num_pages) {
402 uint32_t pages_acked_ptr = (uint32_t) get_cb_tiles_acked_ptr(operand);
403
404 // while the producer (write-side interface) is waiting for space to free up "tiles_pushed" is not changing
405 // "tiles_pushed" is updated by the producer only when the tiles are pushed
406 uint32_t pages_received = get_cb_tiles_received_ptr(operand)[0];
407
408 int32_t free_space_pages;
409 WAYPOINT("CRBW");
410 do {
411 invalidate_l1_cache();
412 // uint16_t's here because Tensix updates the val at tiles_acked_ptr as uint16 in llk_pop_tiles
413 // TODO: I think we could have TRISC update tiles_acked_ptr, and we wouldn't need uint16 here
414 uint16_t pages_acked = (uint16_t)reg_read(pages_acked_ptr);
415#ifdef ARCH_GRAYSKULL
416 // The following test slows down by 5% when removing the barrier
417 // TODO(pgk) investigate GS arbiter WAR in compiler, is this fixing an issue there?
418 // models/experimental/stable_diffusion/tests/test_perf_unbatched_stable_diffusion.py::test_perf_bare_metal
419 volatile uint32_t local_mem_barrier = pages_acked;
420#endif
421 uint16_t free_space_pages_wrap =
422 cb_interface[operand].fifo_num_pages - (pages_received - pages_acked);
423 free_space_pages = (int32_t)free_space_pages_wrap;
424 } while (free_space_pages < num_pages);
425 WAYPOINT("CRBD");
426}

450FORCE_INLINE
451void cb_wait_front(int32_t operand, int32_t num_pages) {
452 uint32_t pages_acked = get_cb_tiles_acked_ptr(operand)[0];
453 uint32_t pages_received_ptr = (uint32_t) get_cb_tiles_received_ptr(operand);
454
455 uint16_t pages_received;
456
457 WAYPOINT("CWFW");
458 do {
459 invalidate_l1_cache();
460 pages_received = ((uint16_t)reg_read(pages_received_ptr)) - pages_acked;
461 } while (pages_received < num_pages);
462 WAYPOINT("CWFD");
463}

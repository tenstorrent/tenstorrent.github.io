635template <typename DType>
636void EnqueueReadBuffer(
637 CommandQueue& cq,
638 Buffer& buffer,
639 std::vector<DType>& dst,
640 bool blocking,
641 tt::stl::Span<const SubDeviceId> sub_device_ids = {}) {
642 dst.resize(buffer.page_size() * buffer.num_pages() / sizeof(DType));
643 EnqueueReadBuffer(cq, buffer, static_cast<void*>(dst.data()), blocking, sub_device_ids);
644}

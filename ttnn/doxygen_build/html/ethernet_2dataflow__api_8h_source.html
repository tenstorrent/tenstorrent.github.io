156void eth_send_bytes(
157 uint32_t src_addr,
158 uint32_t dst_addr,
159 uint32_t num_bytes,
160 uint32_t num_bytes_per_send = 16,
161 uint32_t num_bytes_per_send_word_size = 1) {
162 uint32_t num_bytes_sent = 0;
163 while (num_bytes_sent < num_bytes) {
164 internal_::eth_send_packet(

166 num_bytes_sent += num_bytes_per_send;
167 }
168 erisc_info->channels[0].bytes_sent += num_bytes;
169}

452FORCE_INLINE
453void eth_wait_for_bytes_on_channel_sync_addr(uint32_t num_bytes, volatile eth_channel_sync_t* eth_channel_syncs, uint32_t wait_min = 1000000) {
454 // assert(channel < 4);
455 uint32_t count = 0;
456 uint32_t num_bytes_sent = eth_channel_syncs->bytes_sent;
457 while (num_bytes_sent != num_bytes) {
458 uint32_t received_this_iter = eth_channel_syncs->bytes_sent;
459 if (received_this_iter != num_bytes_sent) {
460 // We are currently in the process of receiving data on this channel, so we just just wait a
461 // bit longer instead of initiating a context switch
462 num_bytes_sent = received_this_iter;
463 } else if (count == wait_min) {
464 count = 0;
465 run_routing();
466 } else {
467 count++;
468 }
469 }
470}

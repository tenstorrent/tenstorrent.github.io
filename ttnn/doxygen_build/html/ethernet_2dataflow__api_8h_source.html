149void eth_send_bytes(
150 uint32_t src_addr,
151 uint32_t dst_addr,
152 uint32_t num_bytes,
153 uint32_t num_bytes_per_send = 16,
154 uint32_t num_bytes_per_send_word_size = 1) {
155 uint32_t num_bytes_sent = 0;
156 while (num_bytes_sent < num_bytes) {
157 internal_::eth_send_packet(

159 num_bytes_sent += num_bytes_per_send;
160 }
161 erisc_info->channels[0].bytes_sent += num_bytes;
162}

444FORCE_INLINE
445void eth_wait_for_bytes_on_channel_sync_addr(
446 uint32_t num_bytes, volatile eth_channel_sync_t* eth_channel_syncs, uint32_t wait_min = 1000000) {
447 // assert(channel < 4);
448 uint32_t count = 0;
449 uint32_t num_bytes_sent = eth_channel_syncs->bytes_sent;
450 while (num_bytes_sent != num_bytes) {
451 uint32_t received_this_iter = eth_channel_syncs->bytes_sent;
452 if (received_this_iter != num_bytes_sent) {
453 // We are currently in the process of receiving data on this channel, so we just just wait a
454 // bit longer instead of initiating a context switch
455 num_bytes_sent = received_this_iter;
456 } else if (count == wait_min) {
457 count = 0;
458 run_routing();
459 } else {
460 count++;
461 }
462 }
463}

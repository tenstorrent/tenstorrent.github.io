116 uint32_t num_bytes,
117 uint32_t num_bytes_per_send = 16,
118 uint32_t num_bytes_per_send_word_size = 1) {
119 uint32_t num_bytes_sent = 0;

124 }
125 erisc_info->channels[0].bytes_sent += num_bytes;
126}

389FORCE_INLINE
390void eth_wait_for_bytes_on_channel(uint32_t num_bytes, uint8_t channel) {
391 // assert(channel < 4);
392 uint32_t count = 0;
393 uint32_t poll_count = 1000000;
394 uint32_t num_bytes_sent = erisc_info->channels[channel].bytes_sent;
395 while (num_bytes_sent == 0) {
396 uint32_t received_this_iter = erisc_info->channels[channel].bytes_sent;
397 if (received_this_iter != num_bytes_sent) {
398 // We are currently in the process of receiving data on this channel, so we just just wait a
399 // bit longer instead of initiating a context switch
400 num_bytes_sent = received_this_iter;
401 } else {
402 count++;
403 if (count > poll_count) {
404 count = 0;
405 run_routing();
406 }
407 }
408 }
